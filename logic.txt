there are 3 components that make the tic tac toe app.

1. players:
    Players are owners of identity.
        what do they own?
            1. symbols: x or o
            2. they can have names
                - player1
                - player2
                - custom names
        how many players do we have?
        - we have exactly two players at any given point of the game.
        -  these players are: 
                xPlayer: 
                    - has name: player1 || customName
                    - has symbol: x
                oPlayer: 
                    - has name: player2 || customName
                    - has symbol: o

            - they only pass the game controller where to put their marker.

2. board: 
    What does the board own?
        - a single spot at any given time can only either be "x" or "o" or empty.
        1. array of marker locations
        2. knows winning combinations
        3. knows the empty and taken spots. 
        4. knows what symbol is at any given spot.
    
    what does the board do?
        1. gives information to the game controller.
                Answers questions such as,
                    1. is the board full,
                    2. is the requested space taken,
                    3. 
        

3. Game controller 
    - this is the coordinator of the game and the decision maker.
    
    responsibilities: 
        1. takes the player's choices
        2 checks if the requested spot if available on the board
        3. places player symbol on the requested spot on board
        4. checks for a win winning combinations
        5. declares a winner
        6. ends the game
        6. cleans the board
        5. checks if the board is full
        5. switches turns between players

function isMoveValid() {
    let board = this.board;
    let move = this.currentPlayerMove.move;
    let status;
    let valid = true;
    let inValid = false;
    if (move >= 0 && move < board.length) {
      if (board[move] === "") {
        status = valid;
      } else {
        status = inValid;
      }
    } else {
      status = inValid;
    }
    return status;
  }

  selectBoardCell() {
      let board = this.board;
      let symbol = this.currentPlayer.symbol;
      let move;
      function getCell() {
        let cell = parseInt(prompt(`Player ${symbol} MOVE: `));
        while (isNaN(cell) || cell < 0 || move > this.board.length) {
          cell = parseInt(prompt(`Player ${symbol} MOVE: `));
        }
        return cell;
      }
      do {
        move = getCell();
      } while (this.board[move] != "");

      return {
        symbol,
        move,
      };
    },
